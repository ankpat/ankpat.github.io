#!/usr/bin/env python3
r"""The Perceptron is the most basic unit of a Neural Network.

The Perceptron provides a mapping from :math:`\mathbb{R}^n\rightarrow\mathbb{R}` by using
a linear basis function for the activity function and a signmoid function for the activation
function.

Given a set of inputs :math:`\{x_1, x_2, \ldots, x_n\}`, a set of weights
:math:`\{w_1, w_2, \ldots, w_n\}`, and a bias :math:`\theta`:

The Activity value is computed as the weighted sum of all of the inputs to the Perceptron
plus a Bias:

.. math:: y = \sum_{i=1}^{n} w_i x_i + \theta

The Activation value is computed using the Sigmoid function with the activity value as the
parameter.

.. math:: v = \frac{1}{1+e^{-y}}

"""
import math
from .Types import WEIGHTS, INPUTS



[docs]class Perceptron:
    """A perceptron takes inputs and computes an activity value based on the inputs
    and specified weights and bias.

    Args:
        number_of_inputs (int): Used soley for sanity checking length of weights list
        weights (WEIGHTS): A list of weights for each input
        bias (float): A constant value added to the the weighted sum of inputs

    """
    def __init__(self, number_of_inputs: int, weights: WEIGHTS, bias: float):
        """Perceptron __init__ method

        Create a new perceptron

        """
        if number_of_inputs != len(weights):
            raise ValueError("Number of inputs and number of weights do not match!")
        self._weights = list(weights)
        self._number_of_inputs = number_of_inputs
        self._bias = bias

    def __getitem__(self, index):
        """Get Item interface to make this class iterable in limited situations

        Note: This treats the bias as a special case and will not be returned here

        Args:
            index (int): Index into the list of weights

        Returns:
            A single weight
        """
        return self._weights[index]

    @property
    def number_of_inputs(self):
        """int: Number of inputs expected by this Perceptron"""
        return self._number_of_inputs

    @property
    def weights(self):
        """WEIGHTS: The weights associated with this Perceptron"""
        return list(self._weights)

    @property
    def bias(self):
        """float: The bias associated with this Perceptron"""
        return self._bias


[docs]    def adjust_weights(self, delta: WEIGHTS):
        r"""Adjust weights of Perceptron by some given factor

        Given a list of deltas, :math:`\{\delta_1, \delta_2, \ldots, \delta_m\}, m \geq n`, the new
        weights are calculated as :math:`w_i + \delta_i`. If m = n+1, then the bias is updated.

        Args:
            delta (WEIGHTS): A list of delta values for each of the weights.
                There must be one delta for each weight. An additional delta can be specified to
                update the bias.

        Returns:
            None
        """
        if len(delta) > len(self._weights)+1:
            raise ValueError("Too many deltas for Perceptron")

        self._weights = [w + d for w, d in zip(self._weights, delta)]

        # Bias weight delta was provided
        if len(delta) == len(self._weights)+1:
            self._bias += delta[-1]



[docs]    def activity(self, inputs: INPUTS):
        r"""Compute the value of the activity function.

        The activity function is computed as the weighted sum of all inputs plus the bias

        .. math:: y = \sum_{i=1}^{n} w_i x_i + \theta

        Arguments:
            inputs (Inputs): A list of input values

        Returns:
            Computed value of activity function
        """
        if len(inputs) != len(self._weights):
            raise ValueError("Number of inputs does not match number of weights for Perceptron")

        result = self._bias
        result += sum(a * b for a, b in zip(inputs, self._weights))

        return result



[docs]    def activation(self, inputs: INPUTS):
        r"""Compute the value of the activation function

        The activation function is computed as the sigmoid function with the activity
        function as its parameter.

        .. math:: v = \frac{1}{1+e^{-y}}

        Arguments:
            inputs (Inputs): A list of input values

        Returns:
            Computed value of the activation function
        """
        activity_value = self.activity(inputs)

        return 1.0/(1.0 + math.pow(math.e, -activity_value))


    def __str__(self):
        """Return string representation of Perceptron

        Arguments:

        Returns:
            String
        """
        return "<Perceptron: %s, %e>"%(self.weights, self._bias)

    def __repr__(self):
        """Return string representation of Perceptron

        Arguments:

        Returns:
            String
        """
        return str(self)
