#!/usr/bin/env python3
"""Neural Network Layer of Perceptrons.
"""
from typing import List

from .Perceptron import Perceptron
from .Types import WEIGHTS, INPUTS

LAYER = List[Perceptron]



[docs]class Layer:
    """A Layer consists of one or more Perceptrons that are all given the same set of inputs.
    The outputs of all Perceptrons in the layer are returned as a single vector.
    """

    def __init__(self, number_of_nodes: int, number_of_inputs: int, weights: WEIGHTS, bias: float):
        """Create a new Neural Network Layer

        Arg:
            number_of_nodes (int): Indicate number of nodes in the node layer
            number_of_inputs (int): Used soley for sanity checking length of weights list
            weights (WEIGHTS): A list of weights for each input
            bias (float): A constant value added to the the weighted sum of inputs
        """
        if number_of_nodes == 0:
            return

        if number_of_inputs != len(weights):
            raise ValueError("Number of inputs and number of weights do not match!")

        self.number_of_inputs = number_of_inputs

        self.nodes = [Perceptron(number_of_inputs, weights, bias) for _ in range(number_of_nodes)]


[docs]    @classmethod
    def make_layer(cls, perceptrons: LAYER):
        """Create a new layer from existing Perceptrons

        Args:
            perceptrons (LAYER): A list defining all perceptrons within this layer

        Returns:
            Layer: An initialized Layer object
        """
        layer_size = len(perceptrons)
        if layer_size == 0:
            raise ValueError("Cannot make an empty layer")
        inputs = perceptrons[0].number_of_inputs
        if not all(perceptron.number_of_inputs == inputs for perceptron in perceptrons):
            raise ValueError("All perceptrons in a layer must take the same number of inputs")

        layer = cls(0, 0, [], 0)

        layer.number_of_inputs = inputs
        layer.nodes = perceptrons

        return layer



    def __getitem__(self, index):
        """Return a node from this layer

        Args:
            index (int): Index into layer nodes

        Returns:
            Returns the node at offset index from the start of the node list
        """
        return self.nodes[index]


[docs]    def feed(self, inputs: INPUTS):
        """Feed inputs through the layer and return list of outputs

        Input are fed to all nodes in the layer.

        Args:
            inputs (INPUTS): A list of input values

        Returns:
            List[float] A list containing the output of each node in this layer
        """
        if self.number_of_inputs != len(inputs):
            raise ValueError("Number of inputs and number of weights do not match!")

        results = [p.activation(inputs) for p in self.nodes]

        return results


    def __str__(self):
        return str(self.nodes)

    def __len__(self):
        return len(self.nodes)
